    ## SOQL Grouped Results in Apex ```java // Id acctId = '0011N00001BuAhQQAV '; // has tasks Id acctId = '0011N00001BuAs7QAF'; // has no tasks AggregateResult[] groupedResults = [ SELECT AccountId, MAX(LastModifiedDate) maxLastModifiedDate FROM Task WHERE Status='Completed' AND AccountId IN (:acctId) GROUP BY AccountId ]; System.debug('Row Count: ' + groupedResults.size()); // if you know we have only one row in the grouped result if (groupedResults.size() > 0) { Object AccountId = groupedResults[0].get('AccountId'); Object maxLastModifiedDate = groupedResults[0].get('maxLastModifiedDate'); System.debug('AccountId: ' + AccountId); System.debug('maxLastModifiedDate: ' + maxLastModifiedDate); } else { System.debug('No max(LastModifiedDate) found for this AccountId: ' + acctId); } // if you know we have multiple rows in the grouped result for (AggregateResult ar : groupedResults) { System.debug('AccountId' + ar.get('AccountId System.debug('maxLastModifiedDate' + ar.get('maxLastModifiedDate')); } ```
